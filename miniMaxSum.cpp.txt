/*
 * Complete the 'miniMaxSum' function below.
 *
 * The function accepts INTEGER_ARRAY arr as parameter.
 */

void miniMaxSum(vector<int> arr) {
    long long minSum=0;
    int n=arr.size();
    long long maxSum=0;
    sort(arr.begin(),arr.end());
    for(int i=0;i<n-1;i++){
        minSum=minSum+arr[i];
    }
    for(int j=n-1;j>0;j--){
        maxSum= maxSum+arr[j];
    }
    cout<<minSum<<" "<<maxSum;


}
